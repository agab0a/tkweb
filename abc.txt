    <script language="JavaScript">
        function tinh() {
            var a = parseInt(document.MT.so1.value);
            var b = parseInt(document.MT.so2.value);
            var c = parseInt(document.MT.so3.value);
            var delta = b * b - 4 * a * c;
            if (a === 0 && b === 0 && c === 0) {
                document.MT.kq.value = "Vô số nghiệm";
            } else if (a === 0 && b === 0 && c !== 0) {
                document.MT.kq.value = "Vô nghiệm";
            } else if (a === 0 && b !== 0) {
                document.MT.kq.value = -c / b;
            } else if (delta >= 0) {
                document.MT.kq.value = (-b + Math.sqrt(delta)) / (2 * a);
                document.MT.kq1.value = (-b - Math.sqrt(delta)) / (2 * a);
            } else {
                document.MT.kq.value = "PT vô nghiệm";
                document.MT.kq1.value = "";
            }
        }
    </script>
